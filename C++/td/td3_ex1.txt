        string nom;

    };


   void etudiant::saisie(){
       cout<<"donnez le nom : "<<endl;

       cout<<"donnez le prenom : "<<endl;
       cin>>prenom;
       cout<<"donnez les notes : "<<endl;
       for(int i=0;i<10;i++){
        cin>>T[i];
       }
    }

  void etudiant::affichage(){
      cout<<"le nom de l'etudiant est "<<nom<<" et son prenom :"<<prenom<<endl;
            for(int i=0;i<10;i++){
                cout<<"les notes sont: "<<T[i]<<endl;
            }
  }

    float etudiant::moyenne(){
        float moy;
        for(int i=0;i<10;i++){
            moy+=T[i];
        }
        return moy/10;
    }

    int etudiant::admis(){
        float a= moyenne();
        if (a>10){
            return 1;
        }
        return 0;
    }


    int etudiant::exae_quo(etudiant e){
      return(moyenne()==e.moyenne());
    }
